!define TABLE(name) class name << (T,#FFAAAA) >>

' ── LEFT COLUMN: Brand hierarchy ──

TABLE(brands) {
    **PK id** : SERIAL
    --
    brand_name : VARCHAR(100) UNIQUE
    created_at : TIMESTAMP
}

TABLE(product_types) {
    **PK id** : SERIAL
    --
    product_type : VARCHAR(150) UNIQUE
    created_at : TIMESTAMP
}

TABLE(categories) {
    **PK id** : SERIAL
    --
    category_name : VARCHAR(100) UNIQUE
    created_at : TIMESTAMP
}

TABLE(ranges) {
    **PK id** : SERIAL
    --
    range_name : VARCHAR(100) UNIQUE
    created_at : TIMESTAMP
}

TABLE(products) {
    **PK id** : SERIAL
    --
    FK brand_id : INTEGER
    FK product_type_id : INTEGER
    FK category_id : INTEGER
    FK range_id : INTEGER
    created_at : TIMESTAMP
}

TABLE(product_formats) {
    **PK id** : SERIAL
    --
    FK product_id : INTEGER
    format : VARCHAR(50)
    packaging : VARCHAR(50)
    created_at : TIMESTAMP
}

' ── CENTER COLUMN: Tracking & Scraping ──

TABLE(product_urls) {
    **PK id** : SERIAL
    --
    FK website_id : INTEGER
    FK product_format_id : INTEGER
    FK store_id : INTEGER [nullable]
    url : VARCHAR(1000)
    is_active : BOOLEAN
    created_at : TIMESTAMP
}

TABLE(raw_staging) {
    **PK id** : SERIAL
    --
    FK product_url_id : INTEGER
    payload : TEXT
    status : VARCHAR(20)
    http_status_code : INTEGER
    error_message : TEXT
    scraped_at : TIMESTAMP
    processed_at : TIMESTAMP
}

TABLE(price_history) {
    **PK id** : SERIAL
    --
    FK product_format_id : INTEGER
    FK website_id : INTEGER
    FK store_id : INTEGER [nullable]
    price : DECIMAL(10,2)
    currency : VARCHAR(10)
    scraped_at : TIMESTAMP
    created_at : TIMESTAMP
}

' ── RIGHT COLUMN: Websites & Stores ──

TABLE(websites) {
    **PK id** : SERIAL
    --
    site_name : VARCHAR(100) UNIQUE
    base_url : VARCHAR(500)
    country : VARCHAR(50)
    created_at : TIMESTAMP
}

TABLE(stores) {
    **PK id** : SERIAL
    --
    FK website_id : INTEGER
    store_code : VARCHAR(50)
    store_name : VARCHAR(200)
    created_at : TIMESTAMP
}

TABLE(exchange_rates) {
    **PK id** : SERIAL
    --
    currency : VARCHAR(10)
    date : DATE
    rate_to_eur : FLOAT
}

TABLE(users) {
    **PK id** : SERIAL
    --
    username : VARCHAR(100) UNIQUE
    password_hash : VARCHAR(255)
    full_name : VARCHAR(200)
    role : VARCHAR(20) [admin|user]
    is_active : BOOLEAN
    created_at : TIMESTAMP
    last_login : TIMESTAMP
}


' ── Layout ──

brands           -[hidden]down- product_types
product_types    -[hidden]down- categories
categories       -[hidden]down- products
products         -[hidden]down- product_formats
product_formats  -[hidden]right- product_urls
product_urls     -[hidden]down-  raw_staging
raw_staging      -[hidden]down-  price_history
product_urls     -[hidden]right- websites
websites         -[hidden]down-  stores
stores           -[hidden]down-  exchange_rates
exchange_rates   -[hidden]down-  users

' ── Relationships ──

brands        "1"    -down-  "0..*" products
product_types "1"    -down-  "0..*" products
categories    "1"    -down-  "0..*" products
products      "1"    -down-  "0..*" product_formats
ranges        "1"    -down-  "0..*" products
product_formats "1"  -right- "0..*" product_urls
product_formats "1"  -right- "0..*" price_history
product_urls  "1"    -down-  "0..*" raw_staging
product_urls  "0..*" -right- "1"    websites
price_history "0..*" -right- "1"    websites
product_urls  "0..*" -right- "0..1" stores
price_history "0..*" -right- "0..1" stores
websites      "1"    -down-  "0..*" stores
